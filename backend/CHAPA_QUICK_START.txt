📋 STEP 4: FRONTEND INTEGRATION
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// When user clicks "Pay with Chapa"
const checkout = async () => {
  const response = await fetch(
    'http://localhost:8000/api/v1/storefront/checkout',
    {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${token}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        items: [
          { product_id: 1, quantity: 2 },
          { product_id: 5, quantity: 1 }
        ],
        payment_method: 'chapa',
        shipping_address: 'Kombolcha, Ethiopia',
        phone: '+251911234567'
      })
    }
  );

  const data = await response.json();

  if (data.success && data.data.payment?.checkout_url) {
    // Redirect to Chapa payment page
    window.location.href = data.data.payment.checkout_url;
  }
};

📋 API ENDPOINTS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Complete Checkout:
   POST /api/v1/storefront/checkout
   
2. Pay for Existing Order:
   POST /api/v1/storefront/orders/{id}/pay
   
3. Check Payment Status:
   GET /api/v1/storefront/orders/{id}/payment-status

📋 PAYMENT FLOW
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. User: Adds items to cart
2. User: Clicks "Checkout with Chapa"
3. Frontend: Calls checkout API
4. Backend: Creates order, initializes Chapa payment
5. Backend: Returns checkout_url
6. Frontend: Redirects to checkout_url
7. User: Completes payment on Chapa
8. Chapa: Sends webhook to backend
9. Backend: Verifies & completes order
10. Chapa: Redirects user back to frontend
11. Frontend: Shows success message

Problem: "Secret key not set"
Solution: Add CHAPPA_SECRET_KEY to .env

Problem: "Payment initialization failed"
Solution: Check API key is valid, run php test-chapa.php

Problem: "Checkout URL not returned"
Solution: Check storage/logs/laravel.log for errors

Problem: "Webhook not working"
Solution: Use ngrok for local testing:
  ngrok http 8000
  Configure webhook URL in Chapa dashboard
